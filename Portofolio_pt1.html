#added columns denoting if climb was successful or if any deaths occurred
expeditions <- expeditions |>
  mutate(climb_success = case_when(
    termination_reason == "Success (main peak)" ~ 1,
    termination_reason == "Success (subpeak)" ~ 1,
    termination_reason == "Success (claimed)" ~ 1,
    .default = 0),
          death_on_exp = if_else(member_deaths + hired_staff_deaths > 0, 1, 0))
            
#joined peaks and expeditions dataset to add height of peaks, dropped unused columns         
peak_expeditions <- expeditions |>
  left_join(peaks |> select(peak_name, height_metres), by = c('peak_name')) |>
  select(-oxygen_used, trekking_agency)


#successful summits creating length of trek to summit from basecamp
successful_expeditions <-  peak_expeditions |>
  filter(climb_success == 1) |>
  mutate(duration_to_summit = as.numeric(highpoint_date - basecamp_date))
#summit rate depends on things like weather, running out of food, turning back for various reasons, aka not reaching the top
peak_expeditions |>
  group_by(peak_name) |>
  summarize(total_expeditions = n(),
            successful_treks = sum(climb_success),
            death_on_trek = sum(death_on_exp)) |>
  mutate(summit_rate = successful_treks/total_expeditions,
         death_rate = death_on_trek/total_expeditions) |>
  filter(total_expeditions >= 50) |>
  arrange(desc(total_expeditions)) |>
  head(10) |>
  
  #top 10 peaks by total # of expeditions
  ggplot(aes(x = fct_reorder(peak_name, total_expeditions, .desc = TRUE), y = total_expeditions, fill = peak_name)) +
  geom_col() +
  scale_fill_brewer(palette = "Set3") +
  labs(
    title = "Most Popular Himalayan Peaks by Total Expeditions (1905-2019)",
    x = "Peak Name",
    y = "Number of Expeditions"
  )  +
  theme_bw() +
  theme(legend.position = "none")
